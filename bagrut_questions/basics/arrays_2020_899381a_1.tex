\begin{minted}{csharp}
public static int Above(int[] arr, int num)
{
    int sum = 0;
    for (int i = 0; i < arr.Length; i++)
    {
        sum += arr[i];
        if (sum > num)
            return i;
    }
    return -1;
}
\end{minted}

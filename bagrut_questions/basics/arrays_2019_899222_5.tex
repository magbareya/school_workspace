public static int Biggest(int[] arr)
{
    int maxIndex = 0;
    int maxValue = arr[0];

    for (int i = 1; i < arr.Length; i++)
    {
        if (arr[i] > maxValue)
        {
            maxValue = arr[i];
            maxIndex = i;
        }
    }

    return maxIndex;
}
\end{minted}
\end{english}

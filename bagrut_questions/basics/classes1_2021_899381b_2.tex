class Program
{
    public static PairOfNums[] Generate(int n)
    {
        PairOfNums[] arr = new PairOfNums[n];
        Random rnd = new Random();
        int cnt = 0;
        while (cnt < n)
        {
            int n1 = rnd.Next(1, 1001);
            int n2 = rnd.Next(1, 1001);
            PairOfNums p = new PairOfNums(n1, n2);
            if (p.EndStart())
            {
                arr[cnt] = p;
                cnt++;
            }
        }
        return arr;
    }
}

class AllNumbers
{
    private int[] arrayNum;

    public int LastOddValue()
    {
        for (int i = this.arrayNum.Length - 1; i >= 0; i--)
        {
            if (this.arrayNum[i] % 2 != 0)
                return this.arrayNum[i];
        }
        return -1;
    }
}
\end{minted}

public static int DigitSum(int num1)
{
    int sum = 0;
    while (num1 > 0)
    {
        sum += (num1 % 10);
        num1 /= 10;
    }
    return sum;
}

public static bool DigitExists(int num, int digit)
{
    while (num > 0)
    {
        if (num % 10 == digit)
            return true;
        num /= 10;
    }
    return false;
}
\end{minted}
\end{english}

int availableSpaces = int.Parse(Console.ReadLine());
while (availableSpaces > 0)
{
    string familyName = Console.ReadLine();
    int familyMembers = int.Parse(Console.ReadLine());

    if (familyMembers > availableSpaces)
    {
        Console.WriteLine(familyName + " no");
    }
    else
    {
        availableSpaces -= familyMembers;
        Console.WriteLine(familyName + " " + (familyMembers * 100));
    }
}
\end{minted}

public static int[] Filter(int[] arr, int num)
{
    int count = 0;
    for (int i = 0; i < arr.Length; i++)
    {
        if (arr[i] != num)
            count++;
    }

    int[] result = new int[count];
    int index = 0;
    for (int i = 0; i < arr.Length; i++)
    {
        if (arr[i] != num)
        {
            result[index] = arr[i];
            index++;
        }
    }
    return result;
}
\end{minted}

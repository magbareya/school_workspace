static bool IsDouble(string s)
{
    if(s.Length % 2 != 0)
        return false;
    int mid = s.Length / 2;
    string firstHalf = s.Substring(0, mid);
    string secondHalf = s.Substring(mid, mid);
    return firstHalf == secondHalf;
}

// Another solution using loops instead of Substring
static bool IsDouble(string s)
{
    if(s.Length % 2 != 0)
        return false;
    int mid = s.Length / 2;
    for(int i = 0; i < mid; i++)
    {
        if(s[i] != s[mid + i])
            return false;
    }
    return true;
}
\end{minted}

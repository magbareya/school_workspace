        name = Console.ReadLine();
        Console.WriteLine(name);
        if(userOffer == 0)
        {
            cnt0++;
            Console.WriteLine(name + " have to pay: " + offer0Price);
        }
        else if(userOffer == 1)
        {
            cnt1++;
            Console.WriteLine(name + " have to pay: " + offer1Price);
        }
        else if(userOffer == 2)
        {
            cnt2++;
            Console.WriteLine(name + " have to pay: " + offer2Price);
        }
    }
}

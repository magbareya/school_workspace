static char GetTwinGender(string firstGender, string secondGender)
{
    if (firstGender == "male" && secondGender == "male")
        return 'm';
    else if (firstGender == "female" && secondGender == "female")
        return 'f';
    else
        return 'v';
}
\end{minted}
\end{english}
\end{boxCode}

\begin{minted}{csharp}
static void Main(string[] args)
{
    int malesCnt = 0, femalesCnt = 0;
    for(int i = 1; i <= 1371; i++)
    {
        string firstGender = Console.ReadLine();
        string secondGender = Console.ReadLine();
        char result = GetTwinGender(firstGender, secondGender);
        if (result == 'm')
            malesCnt+=2;
        else if (result == 'f')
            femalesCnt+=2;
        else if (result == 'v')
        {
            malesCnt++;
            femalesCnt++;
        }
    }
    if (malesCnt > femalesCnt)
        Console.WriteLine("MALES");
    else if (femalesCnt > malesCnt)
        Console.WriteLine("FEMALES");
    else
        Console.WriteLine("EQUAL");
}

\end{minted}

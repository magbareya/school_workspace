public static int[] Multiply(int[] arr1, int[] arr2)
{
    if(arr1.Length > arr2.Length)
    {
        int[] temp = arr1;
        arr1 = arr2;
        arr2 = temp;
    }
    // now we know arr1 is the shorter array
    int m = arr2.Length;
    int k = arr1.Length;
    int[] result = new int[m];
    for (int i = 0; i < k; i++)
    {
        result[i] = arr1[i] * arr2[i];
    }
    for (int i = k; i < m; i++)
    {
        result[i] = arr2[i];
    }
    return result;
}
\end{minted}
\end{english}

\begin{minted}{csharp}
public static bool PosOrder(int[] arr)
{
    int previousPositive = 0;
    for (int i = 0; i < arr.Length; i++)
    {
        if (arr[i] > 0)
        {
            if (arr[i] < previousPositive)
                return false;
            previousPositive = arr[i];
        }
    }
    return true;
}
\end{minted}

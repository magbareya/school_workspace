public static bool Success(int num)
{
    int sum = 0;
    int count6OrAbove = 0;
    bool has10 = false;

    for (int i = 0; i < num; i++)
    {
        int score = int.Parse(Console.ReadLine());
        sum += score;

        if (score >= 6)
            count6OrAbove++;

        if (score == 10)
            has10 = true;
    }

    double average = (double)sum / num;

    if (average > 5.0 || count6OrAbove > (num / 2.0) || has10)
        return true;
    else
        return false;
}
\end{minted}

public class Lesson
{
    private int id;
    private int hh;
    private int mm;
    private int duration;

    public bool IsLater(Lesson other)
    {
        if (this.hh > other.hh)
            return true;
        if (this.hh == other.hh && this.mm > other.mm)
            return true;
        return false;
    }

    public int GetId() { return this.id; }
    public int GetDuration() { return this.duration; }
}

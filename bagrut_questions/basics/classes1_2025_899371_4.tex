public class Transport
{
    private string name;
    private bool toSchool;
    private int num;
    private int day;

    public Transport(string name, int num, int day)
    {
        this.name = name;
        this.num = num;
        this.day = day;
        this.toSchool = true;
    }

    public Transport(int num)
    {
        this.name = "Aviv";
        if (num >= 1 && num <= 50)
            this.num = num;
        else
            this.num = 1;
        this.day = 4;
        this.toSchool = false;
    }

    public string GetName() { return this.name; }
    public bool GetToSchool() { return this.toSchool; }
    public int GetNum() { return this.num; }
    public int GetDay() { return this.day; }
}

class Program
{
    public static int DayReport(Transport[] arr, int day, bool forward)
    {
        int total = 0;
        for (int i = 0; i < arr.Length; i++)
        {
            if (arr[i].GetDay() == day && arr[i].GetToSchool() == forward)
            {
                total += arr[i].GetNum();
            }
        }
        return total;
    }

    public static void MoreForward(Transport[] arr)
    {
        for (int d = 1; d <= 6; d++)
        {
            int forwardCount = DayReport(arr, d, true);
            int backwardCount = DayReport(arr, d, false);
            if (forwardCount > backwardCount)
            {
                Console.WriteLine(d);
            }
        }
    }
}
\end{minted}

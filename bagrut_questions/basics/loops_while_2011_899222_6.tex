double total = 0;
Console.WriteLine("Enter the chips amount:");
int chips = int.Parse(Console.ReadLine());
Console.WriteLine("Enter the juice amount:");
int juice = int.Parse(Console.ReadLine());
while(chips > 5 && juice > 5)
{
    char code = char.Parse(Console.ReadLine());
    int amount = int.Parse(Console.ReadLine());
    if (code == 'a')
    {
        chips -= amount;
        total += (amount * 2.5);
    }
    else if (code == 'b')
    {
        juice -= amount;
        total += (amount * 4.5);
    }
    else if(code == 'c')
    {
        chips -= amount;
        juice -= amount;
        total += (amount * 4.5 + amount * 2.5);
    }
}
Console.WriteLine("Total money: " + total);
Console.WriteLine("Remaining chips: " + chips);
Console.WriteLine("Remaining juice: " + juice);
\end{minted}
\end{english}

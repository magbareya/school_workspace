\begin{minted}{csharp}
static int GetIndexOf(int[] arr, int num)
{
    for(int i = 0; i < arr.Length; i++)
    {
        if (arr[i] == num)
            return i;
    }
    return -1;
}

public static string OriginalText2(string str, int[] arr)
{
    string result = "";
    for (int i = 0; i < str.Length; i++)
    {
        int idx = GetIndexOf(arr, i);
        result += str[idx];
    }

    return result;
}
\end{minted}

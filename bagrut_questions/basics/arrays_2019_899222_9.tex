// Part a: Calculate average satisfaction for a month
public static double MonthlyAverage()
{
    int sum = 0;
    int count = 0;
    int grade = int.Parse(Console.ReadLine());

    while (grade != -1)
    {
        sum += grade;
        count++;
        grade = int.Parse(Console.ReadLine());
    }

    if (count < 100)
        return 0;

    return (double)sum / count;
}

// Part b: Check if employees deserve an excellent trip
public static bool ExcellentTrip(double[] monthlyAverages)
{
    for (int i = 0; i < monthlyAverages.Length - 1; i++)
    {
        if (monthlyAverages[i] > 8 && monthlyAverages[i + 1] > 8)
        {
            return true;
        }
    }

    return false;
}

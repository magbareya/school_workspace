\begin{itemize}
    \item \textbf{$q_0$}: حالة البداية أو لم نجد أي $aa$ حتى الآن.
    \item \textbf{$q_1$}: وجدنا $a$ واحدة فقط (بداية محتملة للسلسلة المطلوبة).
    \item \textbf{$q_2$}: وجدنا $aa$، يجب أن يأتي بعدها $a$ وإلا فإنّ الكلمة مرفوضة (لأنّها احتوت على $aa$ خارج السلسلة $aaa$)
    \item \textbf{$q_3$}: (حالة قابلة) اكتملت الـ $aaa$، أو وجدنا $a$ واحدة بعد اكتمال الشرط سابقاً، يجب أن نرى $b$ بعدها.
    \item \textbf{$q_4$}: (حالة قابلة) وجدنا $b$ بعد اكتمال الشرط.
    \item \textbf{$q_5$}: حالة الخطأ؛ السلسلة احتوت على مخالفة (مثل $aa$ معزولة أو $aaaa$).
\end{itemize}

\end{enumerate}

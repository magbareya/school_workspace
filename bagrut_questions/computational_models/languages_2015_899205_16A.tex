\begin{boxSolution}[سؤال 16a امتحان 899205 سنة 2015]
\begin{enumerate}[itemsep=1.5em, label=\alph*.]
    \item نختار الكلمة $w = 1^i$. عندها نلاحظ أنّ:
    \begin{itemize}
        \item $w_1 \cdot w = 0^i 1^i \in L_1$ لأنّه في الكلمة الناتجة $0^i 1^i$ يظهر الرمز $1$ نفس عدد مرات ظهور الرمز $0$. \\
        أي أنّ: $\#_1(w_1 \cdot w) = i = \#_0(w_1 \cdot w)$.
        \item $w_2 \cdot w = 0^j 1^i \notin L_1$ لأنّه في الكلمة الناتجة $0^j 1^i$ يظهر الرمز $1$ أقل من عدد مرات ظهور الرمز $0$. \\
        لأنّه: $\#_1(w_2 \cdot w) = i < j = \#_0(w_2 \cdot w)$.
    \end{itemize}
\end{enumerate}
\end{boxSolution}

\begin{boxSolution}[سؤال 14b امتحان 899205 سنة 2015]
\begin{enumerate}[itemsep=1.5em, label=\alph*., start=2]
    \item نعرّف اللغتين التاليتين:
    \[ L_1 = \{ a^n | n \geq 0 \} \]
    \[ L_2 = \{ b^m | m \geq 0 \} \]
    عندها، ستكون:
    \[ L_3 = \{ a^n b^m | n, m \geq 0 , n = m\} = \{a^n b^n | n \geq 0 \} \]
    نلاحظ أن $L_1$ و $L_2$ هما لغتان نظاميتان. \\
    بينما اللغة $L_3$ ليست نظامية، لأنّها تعتمد على تذكر عدد غير محدود من الرموز $a$ لمقارنتها بعدد الرموز $b$.

    \item نلاحظ أنّ:
    \begin{enumerate}
        \item $\overline{L_3} = \{ a,b,c \}^+$ أي هي كل التي يمكن تكوينها من الحروف $a$, $b$, و $c$ ما عدا الكلمة الفارغة. \\
        حسب صفات الانغلاق، بما أنّ $L_3$ نظامية، فإنّ $\overline{L_3}$ نظامية بسبب انغلاق اللغات النظامية تحت عملية التكامل.
        \item لذلك: $L_2 \cap \overline{L_3} = \{ c^n | n > 0 \}$ لأنّنا عندما نجري التقاطع مع $\{ a,b,c \}^+$ فإننا نخسر فقط الكلمة الفارغة. \\
        بما أنّ اللغتين $L_2$ و $\overline{L_3}$ نظاميتان، فإنّ تقاطعهما نظامي بسبب انغلاق اللغات النظامية تحت عملية التقاطع.
        \item $R(L_1) = \{ b^k a^n | n,k \geq 0 \}$. \\
        بما أنّ $L_1$ نظامية، فإنّ $R(L_1)$ نظامية بسبب انغلاق اللغات النظامية تحت عملية الانعكاس.
    \end{enumerate}

    حسب تعريف اللغة $L$ فإنّه يمكننا كتابتها بالصورة التالية:
    \[ L = (L_2 \cap \overline{L_3}) \cdot R(L_1) \]
    بما أنّ كل من $L_2 \cap \overline{L_3}$ و $R(L_1)$ نظاميتان، فإنّ $L$ نظامية بسبب انغلاق اللغات النظامية تحت عملية الإلصاق (التجميع).
\end{enumerate}
\end{boxSolution}

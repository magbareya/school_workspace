public static void CheckBrother(int num)
{
    int tens = num / 10;
    int ones = num % 10;
    if (tens == ones || ones == 0)
        Console.WriteLine("No Brother");
    else
        Console.WriteLine(tens + ones*10);
}
\end{minted}
\end{english}

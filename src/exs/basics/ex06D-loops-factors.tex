\begin{enumerate}[itemsep=1.5em]

\item اكتب برنامجًا يستقبل من المستخدم عددًا صحيحا، ويطبع له كل عوامل هذا العدد (أي كل الأعداد التي يقسم عليها هذا العدد).
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
int n = int.Parse(Console.ReadLine());
for (int d = 1; d <= n; d++)
{
    if (n % d == 0)
        Console.WriteLine(d);
}
\end{minted}
\end{english}
\end{boxSolution}
\fi

\item \textbf{للتذكير} عدد أوّلي هو عدد يقسم فقط على نفسه وعلى 1.
    \begin{enumerate}
        \item اكتب عمليّة خارجية باسم \textenglish{IsPrime} تتلقى عددًا صحيحًا وتعيد \textenglish{true} إذا كان هذا العدد أوليًّا، وإلا فإنّها تعيد \textenglish{false}.
        \item في البرنامج الرئيسي، استدعِ العملية مع العدد 17 واطبع النتيجة. ومع العدد 20 واطبع النتيجة.
        \item اكتب عملية خارجية أخرى باسم \textenglish{PrintPrimes} تتلقى عددًا صحيحًا وتطبع كل الأعداد الأولية حتى هذا العدد. \\
                 \textbf{مثلًا:} إذا تلقت العملية العدد 20 فإنّها تطبع: 2، 3، 5، 7، 11، 13، 17، 19.
        \item في البرنامج الرئيسي استدعِ العملية مع الرقم 100.
    \end{enumerate}
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static bool IsPrime(int n)
{
    if (n < 2)
        return false;
    for (int d = 2; d * d <= n; d++)
    {
        if (n % d == 0)
            return false;
    }
    return true;
}

public static void PrintPrimes(int n)
{
    for (int i = 2; i <= n; i++)
    {
        if (IsPrime(i))
            Console.WriteLine(i);
    }
}

private static void Main(string[] args)
{
    Console.WriteLine(IsPrime(17));
    Console.WriteLine(IsPrime(20));
    PrintPrimes(100);
}
\end{minted}
\end{english}
\end{boxSolution}
\clearpage
\fi

\item اكتب عملية تتلقى عددًا صحيحًا، وتطبع كل عوامله \textbf{الأوّلية}. (أي كل الأعداد الأولية التي يقسم عليها هذا العدد) \\
يمكنك الاستعانة بالعملية \textenglish{IsPrime} التي كتبتها في السؤال السابق.
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static void PrintPrimeFactors(int n)
{
    for (int i = 2; i <= n; i++)
    {
        if (n % i == 0 && IsPrime(i))
            Console.WriteLine(i);
    }
}
\end{minted}
\end{english}
\end{boxSolution}
\fi

\item اكتب عملية خارجية باسم \textenglish{IsPrimeSquare} تتلقى عددًا صحيحًا وتعيد \textenglish{true} إذا كان العدد الذي تلقته عبارة عن تربيع عدد أوّلي. وإلا فإنّها تعيد \textenglish{false} \\
\textbf{مثلًا:} تعيد العملية \textenglish{true} إذا تلقّت 25. وتعيد \textenglish{false} إذا تلقت 30.
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static bool IsPrimeSquare(int n)
{
    for (int i = 2; i <= Math.Sqrt(n); i++)
    {
        if (i * i == n && IsPrime(i))
            return true;
    }
    return false;
}
\end{minted}
\end{english}
\end{boxSolution}
\clearpage
\fi

\item اكتب عملية خارجية باسم \textenglish{GCD} تتلقى عددين صحيحين $a$ و $b$، وتعيد القاسم المشترك الأكبر لهما (أكبر عدد يقسم عليه كلا العددين). \\
\textbf{مثلًا:} القاسم المشترك الأكبر للعددين 12 و 15 هو 3، لأنّه أكبر عدد يقسم عليه 12 وأيضًا يقسم عليه 15. \\
\textbf{ملاحظة:} القاسم المشترك الأكبر بالضرورة أصغر أو يساوي \textenglish{Min(a,b)}
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static int GCD(int a, int b)
{
    int min = Math.Min(a, b);
    for (int i = min; i >= 1; i--)
    {
        if (a % i == 0 && b % i == 0)
            return i;
    }
    return 1;
}
\end{minted}
\end{english}
\end{boxSolution}
\else
\clearpage
\fi

\item اكتب عملية خارجية باسم \textenglish{LCM} تتلقى عددين صحيحين $a$ و $b$، وتعيد المقام المشترك الأصغر لهما (أصغر عدد يقسم على العددين معًا). \\
\textbf{مثلًا:} المقام المشترك الأصغر للعددين 6 و 8 هو 24، لأنّه أصغر عدد يقسم على 6 وأيضًا يقسم على 8. \\
\textbf{ملاحظة:} المقام المشترك الأصغر بالضرورة أكبر أو يساوي \textenglish{Max(a,b)} وبالضرورة أصغر من $a \cdot b$
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static int LCM(int a, int b)
{
    int max = Math.Max(a, b);
    for (int i = max; i <= (a*b) ; i++)
    {
        if (i % a == 0 && i % b == 0)
            return i;
    }
}
\end{minted}
\end{english}
\end{boxSolution}
\fi

\item اكتب عملية خارجية باسم \textenglish{IsPerfectNumber} تتلقى عددًا صحيحًا وتعيد \textenglish{true} إذا كان عددًا كاملًا. \\
\textbf{للتذكير:} العدد الكامل هو عدد يساوي مجموع قواسمه باستثناء نفسه. \\
\textbf{مثلًا:} 6 عدد كامل لأن: 1+2+3 = 6
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static bool IsPerfectNumber(int n)
{
    int sum = 0;
    for (int i = 1; i < n; i++)
    {
        if (n % i == 0)
            sum += i;
    }
    return sum == n;
}
\end{minted}
\end{english}
\end{boxSolution}
\fi

\item اكتب عملية خارجية باسم \textenglish{AreAmicable} تتلقى عددين صحيحين وتعيد \textenglish{true} إذا كانا عددين صديقين. \\
\textbf{للتذكير:} عددان صديقان إذا كان مجموع قواسم الأول (بدون نفسه) يساوي الثاني، والعكس صحيح. \\
\textbf{مثلًا:} 220 و 284 عددان صديقان، لأن مجموع قواسم 220 (بدون نفسه) = 284، ومجموع قواسم 284 (بدون نفسه) = 220.
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static int SumOfDivisors(int n)
{
    int sum = 0;
    for (int i = 1; i < n; i++)
    {
        if (n % i == 0)
            sum += i;
    }
    return sum;
}

public static bool AreAmicable(int a, int b)
{
    return SumOfDivisors(a) == b && SumOfDivisors(b) == a;
}
\end{minted}
\end{english}
\end{boxSolution}
\fi

\item اكتب عملية خارجية باسم \textenglish{GCDThreeNumbers} تتلقى ثلاثة أعداد صحيحة $a$ و $b$ و $c$، وتعيد القاسم المشترك الأكبر للأعداد الثلاثة. \\
\textbf{مثلًا:} القاسم المشترك الأكبر للأعداد 12 و 18 و 24 هو 6. \\
\textbf{ملاحظة:} يمكنك الاستعانة بالعملية \textenglish{GCD} التي كتبتها في السؤال السابق.
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static int GCDThreeNumbers(int a, int b, int c)
{
    int gcdAB = GCD(a, b);
    return GCD(gcdAB, c);
}
\end{minted}
\end{english}
\end{boxSolution}
\fi

\item سؤال للتفكير: معطاة معك العملية \textenglish{GCDThreeNumbers} التي كتبتها في السؤال السابق. \\
كيف يمكنك باستخدامها فقط حساب $GCD$ لعددين صحيحين؟

\item اكتب عملية خارجية باسم \textenglish{AreRelativelyPrime} تتلقى عددين صحيحين وتعيد \textenglish{true} إذا كانا أوليين نسبيًا. \\
\textbf{للتذكير:} عددان أوليان نسبيًا إذا كان القاسم المشترك الأكبر لهما يساوي 1. \\
\textbf{مثلًا:} العددان 15 و 28 أوليان نسبيًا لأن القاسم المشترك الأكبر لهما = 1. \\
\textbf{ملاحظة:} يمكنك الاستعانة بالعملية \textenglish{GCD} التي كتبتها في السؤال السابق.
\ifwithsols
\begin{boxSolution}
\begin{english}
\begin{minted}{csharp}
public static bool AreRelativelyPrime(int a, int b)
{
    return GCD(a, b) == 1;
}
\end{minted}
\end{english}
\end{boxSolution}
\fi

\end{enumerate}

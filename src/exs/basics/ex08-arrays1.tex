    % 11. Palindrome Function
    % ------------------------------------------------------
    \item
    اكتب دالة \textenglish{static bool IsPalindrome(int[] arr)} تفحص هل المصفوفة متناظرة (تقرأ من اليمين كاليسار). مثال: $\{1, 2, 1\}$ متناظرة، $\{1, 2, 3\}$ غير متناظرة.
    \ifwithsols
    \begin{boxSolution}
    \begin{english}
    \begin{minted}{csharp}
public static bool IsPalindrome(int[] arr)
{
    // Loop until the middle of the array
    for (int i = 0; i < arr.Length / 2; i++)
    {
        // Compare element from start with element from end
        if (arr[i] != arr[arr.Length - 1 - i])
        {
            return false; // Mismatch found
        }
    }
    return true; // No mismatches found
}
    \end{minted}
    \end{english}
    \end{boxSolution}
    \fi

    % ------------------------------------------------------
    % 12. Shift Right Function
    % ------------------------------------------------------
    \item
    اكتب دالة \textenglish{ShiftRight(int[] arr, int n)} تقوم بإزاحة عناصر المصفوفة \textenglish{n} خطوات لليمين بشكل دائري.
    \ifwithsols
    \begin{boxSolution}
    \begin{english}
    \begin{minted}{csharp}
public static void ShiftRight(int[] arr, int n)
{
    int len = arr.Length;
    int[] temp = new int[len]; // Temporary array

    for (int i = 0; i < len; i++)
    {
        // Calculate new position circularly
        temp[(i + n) % len] = arr[i];
    }

    // Copy back to original array
    for (int i = 0; i < len; i++)
    {
        arr[i] = temp[i];
    }
}
    \end{minted}
    \end{english}
    \end{boxSolution}
    \fi

    % ------------------------------------------------------
    % 13. Is Sorted?
    % ------------------------------------------------------
    \item
    اكتب دالة \textenglish{bool IsSorted(int[] arr)} تعيد \textenglish{true} إذا كانت عناصر المصفوفة مرتبة تصاعدياً (من الأصغر للأكبر)، وإلا تعيد \textenglish{false}.
    \ifwithsols
    \begin{boxSolution}
    \begin{english}
    \begin{minted}{csharp}
public static bool IsSorted(int[] arr)
{
    // Loop up to the second to last element
    for (int i = 0; i < arr.Length - 1; i++)
    {
        // If current element is greater than the next one
        if (arr[i] > arr[i + 1])
            return false;
    }
    return true;
}
    \end{minted}
    \end{english}
    \end{boxSolution}
    \fi

    % ------------------------------------------------------
    % 14. Reverse Array In-Place
    % ------------------------------------------------------
    \item
    اكتب دالة لقلب ترتيب عناصر المصفوفة (Reverse) \textbf{بدون} استخدام مصفوفة إضافية (In-Place Swap).
    \ifwithsols
    \begin{boxSolution}
    \begin{english}
    \begin{minted}{csharp}
public static void ReverseArray(int[] arr)
{
    int start = 0;
    int end = arr.Length - 1;

    while (start < end)
    {
        // Swap logic
        int temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;

        start++;
        end--;
    }
}
    \end{minted}
    \end{english}
    \end{boxSolution}
    \fi

    % 17. Search and Return Index
    % ------------------------------------------------------
    \item
    اكتب دالة \textenglish{int FindIndex(int[] arr, int val)} تعيد رقم الخانة (Index) لأول ظهور للقيمة \textenglish{val}. إذا لم تجدها تعيد -1.
    \ifwithsols
    \begin{boxSolution}
    \begin{english}
    \begin{minted}{csharp}
public static int FindIndex(int[] arr, int val)
{
    for (int i = 0; i < arr.Length; i++)
    {
        if (arr[i] == val)
            return i; // Return index immediately
    }
    return -1; // Not found
}
    \end{minted}
    \end{english}
    \end{boxSolution}

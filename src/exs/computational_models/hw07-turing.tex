\begin{enumerate}[itemsep=2em]
\item صمم آلة تيورينج تقبل اللغة المكونة من كل الكلمات التي تبدأ وتنتهي بنفس الحرف فوق الأبجدية \(\{a, b\}\).

\item صمم آلة تيورينج تقبل اللغة التالية:
\[L = \{a^n b^{2n} \mid n > 0 \}\]

\item صمم آلة تيورينج تقبل اللغة التالية:
\[L = \{a^n b^m \mid n > m > 0 \}\]

\item صمم آلة تيورينج تقبل اللغة التالية:
\[L = \{a^n b^n c^n \mid n > 0\}\]

\end{enumerate}

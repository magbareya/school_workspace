الطباعة

Write

تطبع ويبقى المؤشر على نفس السطر.

WriteLine

تطبع وتنزل سطراً جديداً.

Console.Write("Hello ");
Console.WriteLine("World");
Console.Write("Done");
Output Hello World
Done

C

1

Indexing (الوصول)

نصل للحرف عن طريق رقمه (Index).

string s = "Code";
Console.WriteLine(s[2]); // 'd'

Length (الطول)

عدد الحروف الكلي.

Console.WriteLine(s.Length); // 4
حالة الأحرف (Case)

ToUpper()

تحويل النص بالكامل لحروف كبيرة.

ToLower()

تحويل النص بالكامل لحروف صغيرة.

C# Example
string s = "MoHaMeD";

string upper = s.ToUpper();
string lower = s.ToLower();

Console.WriteLine(upper);
Console.WriteLine(lower);
Output MOHAMED
mohamed
البحث: Contains

تفحص هل النص يحتوي على مقطع معين أو حرف معين.

النتيجة دائماً: True أو False.

// 1. البحث عن نص (String)
string txt = "Hello World";
bool hasWorld = txt.Contains("World");
Console.WriteLine(hasWorld); // True
// 2. البحث عن حرف (Char)
string email = "user@mail.com";
if (email.Contains('@'))
  Console.WriteLine("Valid Email");
تحديد الموقع: IndexOf

تعيد رقم الخانة (Index) لأول ظهور للنص أو الحرف.

إذا لم تجد النص، تعيد -1.

// 1. موقع نص داخل نص
string s = "Banana";
int idx1 = s.IndexOf("nan");
Console.WriteLine(idx1); // 2
// 2. موقع حرف
string date = "2024/05/20";
int idx2 = date.IndexOf('/');
Console.WriteLine(idx2); // 4
القص: Substring (1)

Substring(startIndex)

تقص النص ابتداءً من مكان محدد وحتى النهاية.

"Unhappy"
 0123456
قص حتى النهاية
string word = "Unhappy";

// ابدأ من الخانة 2 وحتى النهاية
string result = word.Substring(2);

Console.WriteLine(result);
Output happy
القص: Substring (2)

Substring(start, length)

تقص النص ابتداءً من مكان محدد وبطول محدد (عدد حروف).

"SmartPhone"
 0123456789
قص بطول محدد
string text = "SmartPhone";

// ابدأ من 2، وخذ 3 حروف
string sub = text.Substring(2, 3);

Console.WriteLine(sub);
Output art
دوال خارجية (Static Functions)

كيف نكتب دوال تتعامل مع النصوص؟

public static string CleanString(string input)
{
  return input.Trim().ToLower();
}
مثال 1: طباعة الحرف الأول والأخير

دالة PrintEnds تستقبل نصاً وتطبع الحرف الأول والأخير منه.

Code & Usage
// 1. تعريف العملية (Function)
public static void PrintEnds(string s)
{
  if (s.Length > 0)
  {
    Console.WriteLine("First: " + s[0]);
    Console.WriteLine("Last: " + s[s.Length - 1]);
  }
}

// 2. الاستدعاء في Main
static void Main(string[] args)
{
  string myText = "Hello";
  PrintEnds(myText); // First: H, Last: o
}
مثال 2: دالة تعيد قيمة (عدد تكرار الحرف)

دالة CountChar تستقبل نصاً وحرفاً، وتعيد عدد مرات ظهوره.

Code & Usage
// 1. تعريف العملية (Function)
public static int CountChar(string str, char c)
{
  int count = 0;
  for (int i = 0; i < str.Length; i++)
  {
    if (str[i] == c) count++;
  }
  return count;
}

// 2. الاستدعاء في Main
static void Main(string[] args)
{
  int res = CountChar("Banana", 'a');
  Console.WriteLine(res); // Output: 3
}
